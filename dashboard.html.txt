    <script>
        let tasks = [];

        const defaultTasks = [
            {id: 0, name: "Oração Matinal", start: "06:00", end: "06:30", done: false, skipped: false},
            {id: 1, name: "Café da Manhã", start: "06:30", end: "07:00", done: false, skipped: false},
            {id: 2, name: "Internet/Ler Notícias", start: "07:00", end: "07:30", done: false, skipped: false},
            {id: 3, name: "Estudo Bíblico", start: "07:30", end: "08:30", done: true, skipped: false},
            {id: 4, name: "Exercícios Físicos", start: "08:30", end: "09:00", done: true, skipped: false},
            {id: 5, name: "Banho e Vestir", start: "09:00", end: "09:30", done: false, skipped: false},
            {id: 6, name: "Planejamento do Dia", start: "09:30", end: "10:00", done: false, skipped: false},
            {id: 7, name: "Almoço", start: "12:00", end: "12:30", done: false, skipped: false},
            {id: 8, name: "Descanso Pós-Almoço", start: "12:30", end: "13:00", done: false, skipped: false},
            {id: 9, name: "Trabalho/Estudos", start: "13:00", end: "17:00", done: false, skipped: false}
        ];

        function loadTasks() {
            // TODO: Integração com Supabase
            // const { data } = await supabase.from('daily_tasks').select('*').eq('user_id', getUserId()).eq('date', getTodayDate());
            // tasks = data || defaultTasks;
            tasks = JSON.parse(localStorage.getItem('ppt_tasks')) || defaultTasks;
        }

        function saveTasks() {
            localStorage.setItem('ppt_tasks', JSON.stringify(tasks));
        }

        function addMinutes(timeStr, mins) {
            const [h, m] = timeStr.split(':').map(Number);
            let newM = m + mins;
            let newH = h + Math.floor(newM / 60);
            newM = newM % 60;
            return `${newH.toString().padStart(2, '0')}:${newM.toString().padStart(2, '0')}`;
        }

        function completeTask(id) {
            const task = tasks.find(t => t.id === id);
            if (task) {
                task.done = true;
                saveTasks();
                renderHoje();
            }
        }

        function skipTask(id) {
            const task = tasks.find(t => t.id === id);
            if (task) {
                task.skipped = true;
                saveTasks();
                renderHoje();
            }
        }

        function extendTask(id) {
            const task = tasks.find(t => t.id === id);
            if (task) {
                task.end = addMinutes(task.end, 10);
                saveTasks();
                renderHoje();
            }
        }

        function renderHoje() {
            const now = new Date();
            const options = { weekday: 'long', day: 'numeric', month: 'long' };
            document.getElementById('dia-title').textContent = `HOJE - ${now.toLocaleDateString('pt-BR', options)}`;

            const pending = tasks.filter(t => !t.done && !t.skipped);
            const doneCount = tasks.filter(t => t.done).length;
            const total = tasks.length;
            const progressPct = ((doneCount / total) * 100).toFixed(0);

            document.getElementById('progress').style.width = progressPct + '%';
            document.getElementById('progress-text').textContent = progressPct + '%';
            document.getElementById('counter').textContent = `${doneCount} de ${total} tarefas concluídas`;

            // AGORA
            const current = pending[0];
            if (current) {
                document.getElementById('agora').innerHTML = `
                    <h3>AGORA</h3>
                    <div class="task-name">${current.name}</div>
                    <div class="time">${current.start} - ${current.end}</div>
                    <div class="buttons">
                        <button class="btn-complete" onclick="completeTask(${current.id})">✓ Concluída</button>
                        <button class="btn-skip" onclick="skipTask(${current.id})">❌ Não Fiz</button>
                        <button class="btn-extend" onclick="extendTask(${current.id})">⏳ +10 min</button>
                    </div>
                `;
            } else {
                document.getElementById('agora').innerHTML = `
                    <h3>AGORA</h3>
                    <div class="completed">🎉 Parabéns! Todas as tarefas do dia foram concluídas!</div>
                `;
            }

            // PRÓXIMA
            const nextTask = pending[1];
            if (nextTask) {
                document.getElementById('proxima').innerHTML = `
                    <h3>PRÓXIMA</h3>
                    <div class="task-name">${nextTask.name}</div>
                    <div class="time">${nextTask.start} - ${nextTask.end}</div>
                    <div class="alert">🔔 5 min de alerta</div>
                `;
            } else {
                document.getElementById('proxima').innerHTML = '<h3>PRÓXIMA</h3><p>Nenhuma tarefa próxima.</p>';
            }

            // APÓS
            const afterTasks = pending.slice(2);
            let afterHtml = afterTasks.map(t => `
                <li>
                    <span class="task-name">${t.name}</span>
                    <span class="time">${t.start} - ${t.end}</span>
                </li>
            `).join('');
            if (afterTasks.length === 0) {
                afterHtml = '<li style="text-align:center; color:var(--gray);">Nenhuma tarefa restante.</li>';
            }
            document.getElementById('apos-list').innerHTML = afterHtml;
        }

        function showPage(pageId) {
            document.querySelector('.nav-item.active').classList.remove('active');
            const activeNav = document.querySelector(`.nav-item[data-page="${pageId}"]`);
            activeNav.classList.add('active');
            document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
            document.getElementById(pageId).classList.add('active');
            if (window.innerWidth < 768) {
                closeSidebar();
            }
            if (pageId === 'hoje') {
                renderHoje();
            }
        }

        function toggleSidebar() {
            document.getElementById('sidebar').classList.toggle('open');
            document.getElementById('overlay').classList.toggle('show');
        }

        function closeSidebar() {
            document.getElementById('sidebar').classList.remove('open');
            document.getElementById('overlay').classList.remove('show');
        }

        function addTask() {
            alert('Adicionar Tarefa - Funcionalidade em desenvolvimento!');
        }

        function logout() {
            localStorage.removeItem('ppt_tasks');
            localStorage.removeItem('user');
            alert('Logout realizado com sucesso!');
            window.location.href = 'login.html';
        }

        // Inicialização
        document.addEventListener('DOMContentLoaded', () => {
            document.getElementById('username').textContent = localStorage.getItem('user') || 'Diego';
            loadTasks();
            renderHoje();

            // Navegação
            document.querySelectorAll('.nav-item').forEach(item => {
                item.addEventListener('click', () => {
                    const pageId = item.dataset.page;
                    showPage(pageId);
                });
            });
        });
